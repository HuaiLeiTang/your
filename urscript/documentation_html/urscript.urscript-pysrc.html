589                  if (yy < epsilon): 
590                      axis = (root_half, 0, root_half) 
591                  else: 
592                      y = math.sqrt(yy) 
593                      axis = (xy/y, y, yz/y)  
594              else:  
595                  # vectors[2][2] is the largest diagonal term so base result on this 
596                  if (zz < epsilon): 
597                      axis = (root_half, root_half, 0) 
598                  else: 
599                      z = math.sqrt(zz) 
600                      axis = (xz/z, yz/z, z) 
601              return Axis_angle(math.pi,axis)             
602      else: 
603          #no singularities 
604          s = math.sqrt(\ 
605              (vectors[1][2] - vectors[2][1]) * (vectors[1][2] - vectors[2][1]) + \ 
606              (vectors[2][0] - vectors[0][2]) * (vectors[2][0] - vectors[0][2]) + \ 
607              (vectors[0][1] - vectors[1][0]) * (vectors[0][1] - vectors[1][0])) 
608          if (math.fabs(s) < 0.001): 
609              #prevent divide by zero, should not happen if vectors are orthonormal 
610              s = 1 
611          angle = math.acos((vectors[0][0] + vectors[1][1] + vectors[2][2] - 1) / 2) 
612          axis = ((vectors[1][2] - vectors[2][1])/s, (vectors[2][0] - vectors[0][2])/s, (vectors[0][1] - vectors[1][0])/s) 
613          return Axis_angle(angle, axis)
